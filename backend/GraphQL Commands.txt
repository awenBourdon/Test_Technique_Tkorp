ON GRAPHQL PLAYGROUND

PERSON

Create a person :

mutation {
  createPerson(createPersonInput: {
    lastName: "John Doe"
    firstName: "Doe updated"
    email: "john@example.com"
    phoneNumber : "555-555-555"
  }) {
    id
    lastName
    firstName
    email
    phoneNumber
  }
}

Get all persons :

query {
  persons{
    id
    firstName
    lastName
    email
    phoneNumber
  }
}

Get a specific person by id : 

query {
  person(id: 1 ) {
    id
    firstName
    lastName
    email
    phoneNumber
  }
}

Delete a person by id : 

mutation {
  deletePerson(id: 1)
}
